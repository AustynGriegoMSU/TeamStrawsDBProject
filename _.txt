    def setup_tables(self):
        """Create tables if they do not exist."""
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS accounts (
                account_id INTEGER PRIMARY KEY AUTOINCREMENT,
                customer_name TEXT NOT NULL,
                branch_name TEXT NOT NULL,
                balance REAL NOT NULL DEFAULT 0.0,
                created_at TEXT NOT NULL
            );
        """)
 
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS transactions (
                transaction_id INTEGER PRIMARY KEY AUTOINCREMENT,
                account_id INTEGER NOT NULL,
                employee_name TEXT NOT NULL,
                branch_name TEXT NOT NULL,
                transaction_type TEXT NOT NULL CHECK(transaction_type IN ('deposit', 'withdraw')),
                amount REAL NOT NULL CHECK(amount > 0),
                balance_before REAL NOT NULL,
                balance_after REAL NOT NULL,
                created_at TEXT NOT NULL,
                FOREIGN KEY (account_id) REFERENCES accounts(account_id)
            );
        """)
 
    def get_account(self, account_id):
        row = self.conn.execute("""
            SELECT account_id, customer_name, branch_name, balance
            FROM accounts
            WHERE account_id = ?
        """, (account_id,)).fetchone()
        return row
 
    def show_account(self, account_id):
        account = self.get_account(account_id)
        if not account:
            print("Account not found.")
            return
 
        print("\n--- Account Info ---")
        print(f"Account ID   : {account[0]}")
        print(f"Customer Name: {account[1]}")
        print(f"Branch       : {account[2]}")
        print(f"Balance      : ${account[3]:.2f}")
 
    def deposit(self, account_id, amount, employee_name, branch_name):
        if amount <= 0:
            print("Deposit amount must be greater than 0.")
            return
 
        account = self.get_account(account_id)
        if not account:
            print("Account not found.")
            return
 
        balance_before = float(account[3])
        balance_after = balance_before + amount
        now = datetime.now().isoformat()
 
        self.conn.execute("""
            UPDATE accounts
            SET balance = ?
            WHERE account_id = ?
        """, (balance_after, account_id))
 
        self.conn.execute("""
            INSERT INTO transactions (
                account_id, employee_name, branch_name, transaction_type,
                amount, balance_before, balance_after, created_at
            )
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            account_id,
            employee_name,
            branch_name,
            "deposit",
            amount,
            balance_before,
            balance_after,
            now
        ))
 
        print(f"Deposit successful. New balance: ${balance_after:.2f}")
 
    def withdraw(self, account_id, amount, employee_name, branch_name):
        if amount <= 0:
            print("Withdrawal amount must be greater than 0.")
            return
 
        account = self.get_account(account_id)
        if not account:
            print("Account not found.")
            return
 
        balance_before = float(account[3])
 
        if amount > balance_before:
            print("Withdrawal denied: insufficient funds.")
            return
 
        balance_after = balance_before - amount
        now = datetime.now().isoformat()
 
        self.conn.execute("""
            UPDATE accounts
            SET balance = ?
            WHERE account_id = ?
        """, (balance_after, account_id))
 
        self.conn.execute("""
            INSERT INTO transactions (
                account_id, employee_name, branch_name, transaction_type,
                amount, balance_before, balance_after, created_at
            )
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            account_id,
            employee_name,
            branch_name,
            "withdraw",
            amount,
            balance_before,
            balance_after,
            now
        ))
 
        print(f"Withdrawal successful. New balance: ${balance_after:.2f}")
 
    def show_transactions(self, account_id):
        rows = self.conn.execute("""
            SELECT transaction_id, transaction_type, amount,
                   employee_name, branch_name, balance_before,
                   balance_after, created_at
            FROM transactions
            WHERE account_id = ?
            ORDER BY transaction_id DESC
        """, (account_id,)).fetchall()
 
        if not rows:
            print("No transactions found for this account.")
            return
 
        print("\n--- Transaction History ---")
        for row in rows:
            print(
                f"ID: {row[0]} | Type: {row[1]} | Amount: ${row[2]:.2f} | "
                f"Employee: {row[3]} | Branch: {row[4]} | "
                f"Before: ${row[5]:.2f} | After: ${row[6]:.2f} | Date: {row[7]}"
            )
 
 
def employee_menu():
    db = BankingDB()
    db.setup_tables()
    db.add_sample_accounts()
 
    print("=== Employee Branch View: Deposit / Withdrawal Management ===")
    employee_name = input("Enter employee name: ").strip()
    branch_name = input("Enter branch name: ").strip()
 
    while True:
        print("\nChoose an option:")
        print("1. View account")
        print("2. Deposit")
        print("3. Withdraw")
        print("4. View account transaction history")
        print("5. Exit")
 
        choice = input("Enter choice: ").strip()
 
        if choice == "1":
            try:
                account_id = int(input("Enter account ID: "))
                db.show_account(account_id)
            except ValueError:
                print("Invalid account ID.")
 
        elif choice == "2":
            try:
                account_id = int(input("Enter account ID: "))
                amount = float(input("Enter deposit amount: "))
                db.deposit(account_id, amount, employee_name, branch_name)
            except ValueError:
                print("Invalid input.")
 
        elif choice == "3":
            try:
                account_id = int(input("Enter account ID: "))
                amount = float(input("Enter withdrawal amount: "))
                db.withdraw(account_id, amount, employee_name, branch_name)
            except ValueError:
                print("Invalid input.")
 
        elif choice == "4":
            try:
                account_id = int(input("Enter account ID: "))
                db.show_transactions(account_id)
            except ValueError:
                print("Invalid account ID.")
 
        elif choice == "5":
            print("Exiting employee branch view.")
            break
 
        else:
            print("Invalid option. Try again.")
 
 
if __name__ == "__main__":
    employee_menu()
